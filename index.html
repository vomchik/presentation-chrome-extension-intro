Друг, что за экстеншен?

Расширение (Extension) - это небольшая программа которая дает возможность значиетельно расширить возможности браузера.

Поехали

  • Открываем Сhrome
  • Вводим chrome://extensions
  • Отмечаем галочку Developer mode

Структура проекта

manifest.json


{
    "manifest_version": 2,
    "name": "My Super Extension",
    "version": "1.0.90",
    "icons": { "16": "16x16.png", "128": "128x128.png" },
    "permissions": [
        "tabs",
        "http://*/*",
        "https://*/*"
    ],
    ...
}

manifest.json


{
    ...,
    "background" : {
        "page": "background.html"
    },
    "content_scripts":[
       {
            "matches": ["http://*/*", "https://*/*"],
            "js": ["frontend.js"]
       }
    ],
    "options_page": "options.html",
    ...
}

manifest.json


{
    ...,
    "web_accessible_resources": [
        "css/main.css",
        "js/main.js"
    ],
    "browser_action": {
        "default_title": "My Super Extension",
        "default_icon" : "16x16.png",
        "default_popup": "login.html"
    }
}
